                                            className="flex h-10 w-10 items-center justify-center rounded-full bg-slate-100 text-slate-600 transition-colors hover:bg-blue-600 hover:text-white"
                                        >
                                            <Icon className="h-5 w-5" />
                                        </a>
                                    ),
                                )}
                            </div>
                        </div>

                        {/* Quote */}
                        <div className="rounded-2xl border border-amber-100 bg-amber-50 p-6">
                            <p className="text-center text-sm font-medium text-amber-800 italic">
                                "Buku adalah jendela dunia. Mari bersama-sama
                                membukanya."
                            </p>
                        </div>
                    </div>
                </div>
            </div>

            {/* Footer */}
            <footer className="border-t border-slate-200 bg-white">
                <div className="mx-auto max-w-6xl px-6 py-12">
                    <div className="grid gap-8 md:grid-cols-3">
                        <div>
                            <div className="mb-4 flex items-center gap-2">
                                <BookOpen className="h-6 w-6 text-blue-600" />
                                <span className="text-lg font-bold">
                                    {profile.nama_perpus}
                                </span>
                            </div>
                            <p className="text-sm leading-relaxed text-slate-500">
                                Membangun literasi, memperkuat komunitas, dan
                                membuka wawasan untuk generasi yang lebih baik.
                            </p>
                        </div>
                        <div>
                            <h4 className="mb-4 font-semibold text-slate-900">
                                Tautan Cepat
                            </h4>
                            <ul className="space-y-2 text-sm text-slate-500">
                                <li>
                                    <a href="#" className="hover:text-blue-600">
                                        Katalog Buku
                                    </a>
                                </li>
                                <li>
                                    <a href="#" className="hover:text-blue-600">
                                        Pendaftaran Anggota
                                    </a>
                                </li>
                                <li>
                                    <a href="#" className="hover:text-blue-600">
                                        Jadwal Kegiatan
                                    </a>
                                </li>
                                <li>
                                    <a href="#" className="hover:text-blue-600">
                                        Galeri Foto
                                    </a>
                                </li>
                            </ul>
                        </div>
                        <div>
                            <h4 className="mb-4 font-semibold text-slate-900">
                                Lokasi
                            </h4>
                            <p className="text-sm text-slate-500">
                                {profile.alamat}
                            </p>
                            <p className="mt-2 text-sm text-slate-500">
                                Telp: {profile.kontak}
                            </p>
                        </div>
                    </div>
                    <div className="mt-8 border-t border-slate-100 pt-8 text-center text-sm text-slate-400">
                        {new Date().getFullYear()} {profile.nama_perpus}.
                        Dibangun untuk masyarakat.
                    </div>
                </div>
            </footer>

            {/* Modal Detail Laporan */}
            {selectedActivity && selectedActivity.laporan && (
                <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/50 p-4 backdrop-blur-sm">
                    <div className="w-full max-w-lg rounded-2xl bg-white p-6 shadow-2xl">
                        <div className="mb-4 flex items-center justify-between">
                            <h3 className="text-lg font-bold text-slate-900">
                                Detail Laporan
                            </h3>
                            <button
                                onClick={() => setSelectedActivity(null)}
                                className="rounded-full p-1 hover:bg-slate-100"
                            >
                                <X className="h-5 w-5 text-slate-500" />
                            </button>
                        </div>
                        <div className="space-y-4">
                            <div>
                                <p className="text-sm font-medium text-slate-500">
                                    Kegiatan
                                </p>
                                <p className="text-base font-semibold text-slate-900">
                                    {selectedActivity.nama_kegiatan}
                                </p>
                            </div>
                            <div>
                                <p className="text-sm font-medium text-slate-500">
                                    Tanggal Kegiatan
                                </p>
                                <p className="text-sm text-slate-700">
                                    {formatDate(
                                        selectedActivity.tanggal_kegiatan,
                                    )}
                                </p>
                            </div>
                            <div>
                                <p className="text-sm font-medium text-slate-500">
                                    Status Laporan
                                </p>
                                <span
                                    className={`inline-flex items-center gap-1 rounded-full px-3 py-1 text-sm font-medium ${
                                        selectedActivity.laporan.status ===
                                        'Selesai'
                                            ? 'bg-green-100 text-green-700'
                                            : 'bg-amber-100 text-amber-700'
                                    }`}
                                >
                                    {selectedActivity.laporan.status ===
                                    'Selesai' ? (
                                        <CheckCircle2 className="h-4 w-4" />
                                    ) : (
                                        <AlertCircle className="h-4 w-4" />
                                    )}
                                    {selectedActivity.laporan.status}
                                </span>
                            </div>
                            <div>
                                <p className="text-sm font-medium text-slate-500">
                                    Tanggal Pembuatan Laporan
                                </p>
                                <p className="text-sm text-slate-700">
                                    {formatTimestamp(
                                        selectedActivity.laporan.tanggal_buat,
                                    )}
                                </p>
                            </div>
                            <div className="rounded-lg border border-slate-100 bg-slate-50 p-4">
                                <p className="mb-2 text-sm font-medium text-slate-500">
                                    Isi Laporan
                                </p>
                                <p className="text-sm leading-relaxed whitespace-pre-wrap text-slate-700">
                                    {selectedActivity.laporan.isi_laporan}
                                </p>
                            </div>
                        </div>
                        <button
                            onClick={() => setSelectedActivity(null)}
                            className="mt-6 w-full rounded-xl bg-blue-600 py-3 text-sm font-semibold text-white hover:bg-blue-700"
                        >
                            Tutup
                        </button>
                    </div>
                </div>
            )}
        </div>
    );
};
